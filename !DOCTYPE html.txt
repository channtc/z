    <script>
        function createStar() {
            const star = document.createElement('div');
            star.classList.add('star');

            // Kích thước ngẫu nhiên cho ngôi sao
            const size = Math.random() * 20 + 10 + 'px';
            star.style.width = size;
            star.style.height = size;

            // Vị trí xuất phát: Giữa màn hình (sau trái tim)
            star.style.left = '50%';
            star.style.top = '50%';

            // Tính toán hướng bay ngẫu nhiên
            const angle = Math.random() * Math.PI * 2; // Góc ngẫu nhiên 0-360 độ
            const velocity = Math.random() * 200 + 100; // Khoảng cách bay
            
            const tx = Math.cos(angle) * velocity + 'px';
            const ty = Math.sin(angle) * velocity + 'px';

            star.style.setProperty('--tx', tx);
            star.style.setProperty('--ty', ty);

            // Màu sắc ngẫu nhiên (Vàng, Trắng, Cam)
            const colors = ['#ffd700', '#ffffff', '#ffaa00', '#00ffff'];
            star.style.background = colors[Math.floor(Math.random() * colors.length)];

            // Thời gian bay ngẫu nhiên
            const duration = Math.random() * 1 + 0.5 + 's';
            star.style.animation = `fly ${duration} linear forwards`;

            document.body.appendChild(star);

            // Xóa ngôi sao sau khi bay xong để nhẹ máy
            setTimeout(() => {
                star.remove();
            }, 1500);
        }

        // Tạo mưa sao liên tục
        setInterval(createStar, 50); // Cứ 50ms tạo 1 ngôi sao
    </script>
